import React, { Component } from 'react';
import { Text, View, ScrollView, TouchableOpacity, Dimensions, Linking } from 'react-native';
import { WebView } from 'react-native-webview';
import _ from 'lodash';
import email from 'react-native-email';
import Icon from 'react-native-vector-icons/FontAwesome';
import IconE from 'react-native-vector-icons/Entypo';
 
import {playstorelink} from "../../oauth";
 
export default class Feedback extends Component {
 
  sendMail() {
    email(['joan.marc.pastor@est.fib.upc.edu'], {
      subject: 'FIB APP - Feedback & Bugs',
    }).catch(() => alert('Error obrint el client de correu... Pots enviar el teus comentaris a joan.marc.pastor@est.fib.upc.edu'));
  }
 
  redirectToPlayStore() {
    Linking.openURL(playstorelink);
  }
 
  render() {
    var { width, height } = Dimensions.get('window');
    return (
      <View style={{flex:1, alignItems:'center', backgroundColor:'#fefefe', padding:10}}>
          <View style={{width:width-20, height:height/2-60}}>
            <WebView  textZoom={100} 
            scalesPageToFit={false}
            source={{
              html: 
              '<style>p{text-align:justify}</style>'+
              '<p>Siéntete libre de enviarme un correo si algo no se comporta como esperabas o si echas en falta alguna funcionalidad. Adjunta capturas de pantalla de todo lo posible.</p>'+
              '<p>Trataré de solucionar errores críticos. Todo lo demás dependerá de mi disponibilidad.</p>'
            }}/>
          </View>
 
            <TouchableOpacity onPress={this.sendMail}>
              <View style={{marginTop:30, width:220, flexDirection:'row', alignItems:'center', justifyContent:'center', backgroundColor:'#B40404', borderRadius:5, padding:8}}>
                <Icon name="envelope-o" color='#fff' size={25}/>
                <Text style={{marginLeft:10, fontFamily: 'Arial', fontSize: 16, fontWeight:'bold', color:'#fff'}}>{'  Envíame un correo!  '}</Text>
              </View>
            </TouchableOpacity>
 
            <TouchableOpacity onPress={this.redirectToPlayStore}>
              <View style={{marginTop:30, width:220, flexDirection:'row', alignItems:'center', justifyContent:'center', backgroundColor:'#00671D', borderRadius:5, padding:8}}>
                <IconE name="google-play" color='#fff' size={25}/>
                <Text style={{marginLeft:10, fontFamily: 'Arial', fontSize: 16, fontWeight:'bold', color:'#fff'}}>{'  Valora la App  ❤   '}</Text>
              </View>
            </TouchableOpacity>
 
            <Text style={{position:'absolute', bottom:20}}>Creada por Joan Marc Pastor con la API del Racó</Text>
        </View>
    );
  }
}
